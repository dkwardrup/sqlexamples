12.   SELECT bookings.facid, SUM(bookings.slots) 
FROM cd.bookings
WHERE starttime BETWEEN '12-09-01' AND '12-09-30'
GROUP BY facid
ORDER BY SUM(slots) DESC;  

13. SELECT facid, SUM(slots) 
FROM cd.bookings
GROUP BY facid
HAVING SUM(slots) > 1000
ORDER BY facid;  

14. SELECT cd.bookings.starttime, cd.facilities.name 
FROM cd.bookings
LEFT JOIN cd.facilities 
ON cd.bookings.facid = cd.facilities.facid
WHERE name LIKE '%Tennis Court%' 
AND starttime >= '2012-09-21' AND starttime < '2012-09-22'
ORDER BY starttime;  

15. SELECT cd.bookings.starttime, cd.members.surname, cd.members.firstname 
FROM cd.bookings
LEFT JOIN cd.members
ON cd.members.memid = cd.bookings.memid
